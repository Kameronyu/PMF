Create `tools/asset/probe_video.py`. Use the bundled ffmpeg (no system ffmpeg). Seed (paste, then harden the parser):

```python
import sys, os, re, json, subprocess
from imageio_ffmpeg import get_ffmpeg_exe

def probe_video(video):
    r = subprocess.run([get_ffmpeg_exe(), "-i", video], capture_output=True, text=True)
    s = r.stderr  # ffmpeg prints stream info to stderr (and exits non-zero with no output file — expected)
    out = {"bytes": os.path.getsize(video)}
    m = re.search(r"Duration:\s*(\d+):(\d+):(\d+\.\d+)", s)
    if m:
        h, mn, sec = int(m.group(1)), int(m.group(2)), float(m.group(3))
        out["duration_s"] = round(h*3600 + mn*60 + sec, 2)
    v = re.search(r"Video:\s*([\w0-9]+).*?,\s*(\d+)x(\d+).*?,\s*([\d.]+)\s*fps", s)
    if v:
        out["codec"] = v.group(1); out["w"] = int(v.group(2)); out["h"] = int(v.group(3)); out["fps"] = float(v.group(4))
        out["aspect"] = f"{out['w']}:{out['h']}"
    out["has_audio"] = bool(re.search(r"Stream.*Audio:", s))
    out["oversize"] = out["bytes"] > 150 * 1024 * 1024
    return out

if __name__ == "__main__":
    print(json.dumps(probe_video(sys.argv[1])))
```

Create `tools/asset/sample_montage.py`. Brick 3v: extract frames at 5fps with the bundled ffmpeg, then montage them into timestamped contact sheets with PIL. Seed (paste, then extend per below):

```python
import sys, os, glob, json, shutil, subprocess
from imageio_ffmpeg import get_ffmpeg_exe
from PIL import Image, ImageDraw
FFMPEG = get_ffmpeg_exe()

def sample(video, outdir, fps=5):
    os.makedirs(outdir, exist_ok=True)
    subprocess.run([FFMPEG, "-i", video, "-vf", f"fps={fps}", os.path.join(outdir, "f%04d.jpg")],
                   check=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)

def montage(frames_dir, sheet_dir, vid_id, fps=5, cols=5, rows=6, cell=360):
    os.makedirs(sheet_dir, exist_ok=True)
    frames = sorted(glob.glob(os.path.join(frames_dir, "f*.jpg")))
    per = cols * rows                       # 30 frames/sheet = 6s at 5fps
    sheets = []
    for s in range((len(frames) + per - 1) // per):
        chunk = frames[s*per:(s+1)*per]
        sheet = Image.new("RGB", (cols*cell, rows*cell), "black"); d = ImageDraw.Draw(sheet)
        for i, fp in enumerate(chunk):
            thumb = Image.open(fp); thumb.thumbnail((cell, cell))
            x, y = (i % cols)*cell, (i // cols)*cell
            sheet.paste(thumb, (x, y))
            gi = s*per + i; t = gi / fps                      # global frame index -> time
            ts = f"{int(t//60):02d}:{t%60:05.2f}"            # mm:ss.ff
            d.rectangle([x+2, y+2, x+78, y+18], fill="black") # legibility backing for the label
            d.text((x+4, y+4), ts, fill="yellow")
        sheet_path = os.path.join(sheet_dir, f"{vid_id}-s{s+1:02d}.jpg")
        sheet.save(sheet_path, "JPEG", quality=85); sheets.append(sheet_path)
    return sheets, len(frames)
```
