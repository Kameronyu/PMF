```javascript
'use strict';
// fanout-path.js — STORE-04: per-cell fan-out filename rule + canonical path-segment
// sanitizer (path-traversal safe). Shared lib: store/version/receipt bricks import this.
// Rule: <axis-a>__<axis-b>.json — keys joined by DOUBLE underscore (single `_` can appear
// inside a value), each key sanitized to [a-z0-9._-]. `<base>-vN` survives the filter.

function sanitizePathSegment(raw) {
  return String(raw)
    .toLowerCase()
    .replace(/[^a-z0-9._-]/g, '')
    .replace(/\.\.+/g, '');   // collapse residual dots — block traversal
}

function buildFanoutName(...keys) {
  return keys.map(sanitizePathSegment).join('__') + '.json';
}

module.exports = { buildFanoutName, sanitizePathSegment };
```

This is the STORE-04 deliverable. Do NOT couple it to any step internals (Research Q2). The `__` separator is mandatory (single `_` appears inside values like `qualifying_creatives`). Collisions are impossible because keys are closed-vocabulary cell coordinates.
