```javascript
const crypto = require('crypto');
function hashInputs(inputPaths) {
  const h = crypto.createHash('sha256');
  for (const p of [...inputPaths].sort()) {            // sort → order-independent
    h.update(p); h.update('\0');
    if (fs.existsSync(p)) h.update(fs.readFileSync(p)); // hash bytes when the file exists
  }
  return h.digest('hex');
}
```
