function runStep(stepId, space, opts) {
  const m = loadManifest(stepId, opts.manifestDir);   // CTRL-11
  preflight(m, space);                                // P1 CTRL-03 â€” missing read -> REFUSE + exit 3
  planPrint(m);                                       // P2 CTRL-04 â€” declare DAG before running
  const ctx = assembleContext(m, space);              // P3 CTRL-05 â€” embed reads[] bytes
  let attempts = 0, verdict, outputs;
  do {
    outputs = spawnWaves(m, ctx, space, attempts);    // P4 CTRL-06 â€” waves <=5; mock-emit; unique id per attempt
    verdict = validate(m, outputs);                   // P5 CTRL-07 â€” EXPLICIT route.js / manifest.validator
    attempts++;
  } while (!verdict.ok && attempts <= 2);             // bounded re-spawn <=2
  if (!verdict.ok) escalate(stepId, verdict);         // -> operator, exit!=0
  const receipt = storeAndReceipt(m, space, verdict, opts); // P6 CTRL-08 â€” space-version + receipt-write (unique id)
  operatorGate(m, receipt, opts);                     // P7 CTRL-09 â€” block / --smoke auto-approve; log in receipt
  return { ok:true, receipt };
}
function runAll(space, opts) {                         // CTRL-02
  for (const id of parsePipeline(opts.pipeline || 'pipeline.yaml')) runStep(id, space, opts);
}
function parsePipeline(file) {                         // CTRL-10 â€” 6-line flat-list splitter, zero-dep
  return require('fs').readFileSync(file,'utf8').split('\n')
    .map(l=>l.trim()).filter(l=>l.startsWith('- ')).map(l=>l.slice(2).trim()).filter(Boolean);
}
